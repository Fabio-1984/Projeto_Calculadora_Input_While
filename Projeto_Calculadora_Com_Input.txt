def soma(num1, num2):
   resultado = num1 + num2
   return resultado

def subtracao(num1, num2):
    resultado = num1 - num2
    return resultado

def multiplicacao(num1, num2):
    resultado = num1 * num2
    return resultado    

def divisao(num1, num2):
    resultado = num1 / num2
    return resultado  


while True: 

 print("1 - Para somar")
 print("2 - Para subtrair")
 print("3 - Para multiplicar")
 print("4 - Para dividir")
 print("0 - Para sair")   
 operacao = int(input("Digite qual das operações deseja realizar: "))
 
 if operacao == 1:
   n1 = float(input("Digite o primeiro número: "))
   n2 = float(input("Digite o segundo número: "))
   total = soma(n1, n2)
   print(total)
   

 elif operacao == 2:
   n1 = float(input("Digite o primeiro número: "))
   n2 = float(input("Digite o segundo número: "))
   total = subtracao(n1, n2)
   print(total)

 elif operacao == 3:
   n1 = float(input("Digite o primeiro número: "))
   n2 = float(input("Digite o segundo número: "))
   total = multiplicacao(n1, n2)
   print(total)

 elif operacao == 4:
   n1 = float(input("Digite o primeiro número: "))
   n2 = float(input("Digite o segundo número: "))
   total = divisao(n1, n2)
   print(total)

 elif operacao > 4:
   print("Opção inválida!")    

 else:
   print("Você saiu do programa com sucesso!")
   break
